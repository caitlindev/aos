employeeLookup.factory("EmployeeLookupService", ->
  _subscribers = []
  selectedEmployees = []

  return {
    subscribe: (cb) ->
      _subscribers.push(cb)

    setCode: (code) ->
      selectedEmployees = code
      angular.forEach _subscribers, (cb) ->
        cb(selectedEmployees)
  }
)
